#include

using namespace std;

int main()
{
int no_1 = 0;
int no_2 = 0;
cout << "Welcome to my program of mathematical operations" << endl;
cout << "Enter two numbers you want to play with :" << endl;
cin >> no_1 >> no_2;
cout << "The operations are done below :" << endl;
int sub = no_1 - no_2;
int add = no_1 + no_2;
int mul = no_1 * no_2;
float div = no_1 / (float)no_2;
int mod = no_1 % no_2;

cout << no_1 << " - " << no_2 << " = " << sub << endl;
cout << no_1 << " + " << no_2 << " = " << add << endl;
cout << no_1 << " * " << no_2 << " = " << mul << endl;
cout << no_1 << " / " << no_2 << " = " << div << endl;
cout << no_1 << " % " << no_2 << " = " << mod << endl;
return 0;
}
